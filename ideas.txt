foo = Foo();
foo.someStaticMember;
Foo.someStaticMember;
Foo.__staticType__ == foo.__type__;
someHash[Foo.__staticType__] = bar;

/* Type name resolves to actual type object, static member access uses different operator */
foo = Foo();
foo.someStaticMember;
Foo::someStaticMember;
Foo == foo.__type__;
someHash[Foo] = bar;
lang.Int /* lang is an object instance, not a class, so . is used, not :: */
Int::__init__(this, 3);
